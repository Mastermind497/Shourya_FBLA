111  public static Chart monthLineGraph(String title, List<Event> events) {

124  String[] xAxisLabels = Event.getMonthsWithYear(events.get(0).getDate(), events.get(events.size() - 1).getDate());
125  XAxis xAxis = configuration.getxAxis();
126  xAxis.setTitle("Months");
127  xAxis.setCategories(xAxisLabels);
128  configuration.addxAxis(xAxis);
129 
130  //The Data for the Chart
131  ListSeries series = new ListSeries("Months");
132  List<Number> hours = new ArrayList<>(xAxisLabels.length);
133  //Creates a list of hours with all of the hours allocated such that it matches with the month labels
134  int currentMonth = events.get(0).getMonth();
135  int currentYear = events.get(0).getYear();
136  double sum = 0;
137  for (int i = 0; i < events.size(); i++) {
138  Event e = events.get(i);
139  if (e.getMonth() == currentMonth && e.getYear() == currentYear) {
140  sum += e.getHours();
141  } else {
142  hours.add(sum);
143  sum = 0;
144  currentMonth++;
145  if (currentMonth > 12) {
146  currentMonth = 0;
147  currentYear++;
148  }
149  i--;
150  }
151  }
152  //Last Sum because it doesn't get added in the loop
153  hours.add(sum);
154  series.setData(hours);
155  configuration.addSeries(series);
156 
157  return chart;

218  private static TreeSeries createTreeSeries(List<StudentData> studentDataList, String chartType) {
219  TreeSeries series = new TreeSeries();
220 
221  int low, med;
222 
223  switch (chartType) {
224  case YEAR_CHART:
225  case ALL_TIME_CHART:
226  low = 25;
227  med = 75;
228  break;
229  default:
230  low = 5;
231  med = 10;
232  break;
233  }
234 
235  TreeSeriesItem hoursLow = new TreeSeriesItem("HL", low + " Or Less Hours");
236  //Red
237  hoursLow.setColorIndex(5);
238 
239  TreeSeriesItem hoursMed = new TreeSeriesItem("HM", med + " Or Less Hours");
240  //Orange
241  hoursMed.setColorIndex(3);
242 
243  TreeSeriesItem hoursHigh = new TreeSeriesItem("HH", "More than " + med + "hours");
244  //Green
245  hoursHigh.setColorIndex(2);
246  series.addAll(hoursLow, hoursMed, hoursHigh);
247 
248  for (StudentData student : studentDataList) {
249  if (student.getCommunityServiceHours() <= low) {
250  series.add(new TreeSeriesItem(student.toString(), hoursLow, round(student.getCommunityServiceHours())));
251  } else if (student.getCommunityServiceHours() <= med) {
252  series.add(new TreeSeriesItem(student.toString(), hoursMed, round(student.getCommunityServiceHours())));
253  } else {
254  series.add(new TreeSeriesItem(student.toString(), hoursHigh, round(student.getCommunityServiceHours())));
255  }
256  }
257 
258  return series;
259  }
