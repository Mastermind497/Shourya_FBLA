101  private String getHoursRemaining(StudentData student) {
102  String category = student.getCommunityServiceCategory();
103  double hours = student.getCommunityServiceHours();
104  if (category.contains("50")) {
105  if (hours >= 50) return "Goal Reached! Next Goal: CSA Service (200 Hours)";
106  else return (50 - hours) + " Hours To Go";
107  } else if (category.contains("200")) {
108  if (hours >= 200) return "Goal Reached! Next Goal: CSA Achievement (500 Hours)";
109  else return (200 - hours) + " Hours To Go";
110  } else {
111  assert category.contains("500");
112  if (hours >= 500) return "Completed the Highest Goal! Congratulations!";
113  else return (500 - hours) + "To Go";
114  }
115  }
